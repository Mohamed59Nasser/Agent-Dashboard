function submitQuizAnswers() { let score = 0; let answersReview = []; activeQuizQuestions.forEach((q, i) => { const sel = document.querySelector(`input[name="q_${i}"]:checked`); const selectedAnswer = sel ? parseInt(sel.value) : null; const isCorrect = selectedAnswer === q.correct; if(isCorrect) score++; // تلوين الإجابة الصحيحة const correctLbl = document.getElementById(`label_${i}_${q.correct}`); if(correctLbl) { correctLbl.classList.add('correct-ans'); } // لو المستخدم اختار الصح if(sel && selectedAnswer == q.correct) { const userCorrect = document.getElementById(`label_${i}_${selectedAnswer}`); if(userCorrect) { userCorrect.classList.add('correct-ans'); } } // لو المستخدم اختار غلط if(sel && !isCorrect) { const wrongLbl = document.getElementById(`label_${i}_${selectedAnswer}`); if(wrongLbl) { wrongLbl.classList.add('wrong-ans'); } } // حفظ تفاصيل الإجابات answersReview.push({ question: q.question, selected: selectedAnswer, correct: q.correct, options: q.opts, status: isCorrect ? "صح" : "غلط" }); }); let res = `${score} / ${activeQuizQuestions.length}`; fetch(`${firebaseBaseUrl}/scores.json`, { method: 'POST', body: JSON.stringify({ username: currentUser.username, fullName: currentUser.fullName, quizName: activeQuizName, score: res, answers: answersReview, timestamp: new Date().toLocaleString('ar-EG') }) }) .then(() => { let reviewHTML = `
✅ نتيجتك النهائية: ${res}
`; answersReview.forEach((a, index) => { reviewHTML += `

سؤال ${index + 1}

${a.question}

حالتك: ${a.status}

إجابتك: ${ a.selected ? a.options[a.selected - 1] : 'لم يتم الاختيار' }

الإجابة الصحيحة: ${a.options[a.correct - 1]}

`; }); document.getElementById( 'quiz-questions-container' ).innerHTML = reviewHTML; document.getElementById( 'submit-quiz-btn' ).style.display = "none"; }); } function loadScoresAdmin() { fetch(`${firebaseBaseUrl}/scores.json`) .then(res => res.json()) .then(scores => { const tbody = document.getElementById( 'admin-scores-list' ); tbody.innerHTML = ''; if(!scores) { tbody.innerHTML = ` لا توجد نتائج مسجلة. `; return; } Object.keys(scores).forEach(id => { let s = scores[id]; tbody.innerHTML += ` ${s.fullName || s.username} ${s.quizName} ${s.score} ${s.timestamp} `; }); }); } function showScoreDetails(scoreId) { fetch( `${firebaseBaseUrl}/scores/${scoreId}.json` ) .then(res => res.json()) .then(data => { let html = `

تفاصيل اختبار ${data.fullName}

`; data.answers.forEach((a, i) => { html += `

س${i+1}: ${a.question}

${a.status}

إجابته: ${ a.selected ? a.options[a.selected - 1] : 'لم يجب' }

الإجابة الصحيحة: ${a.options[a.correct - 1]}

`; }); document.getElementById( 'info-box-title' ).innerText = "تفاصيل نتائج الاختبار"; document.getElementById( 'info-content-area' ).innerHTML = html; switchView('info-display-box'); }); }
